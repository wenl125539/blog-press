虚拟滑动

一、为什么使用虚拟滚动

  • 当数据量过大的时候，无法一次性渲染太多的 DOM 元素
  • 每一次滚动事件将会让对应 DOM 中的所有元素重新渲染

别让浏览器一次性渲染这么多元素，这里通常会对应三种做法来减少元素渲染。

  1. 数据分页
  • 好处是每次只渲染一部分（比如 10 条），下一页在重新渲染另一部分
  • 采用sql分页的做法来分割数据,减少一次性渲染的行数，但一般是后台PC端的分页显示方式

  • 【不适用于手机】

  1. 无限滚动
  • 好处是每次只渲染一部分（比如 10 条），等剩余部分滚动到可见区域，就再渲染另一部分。
  • 同样采用sql分页的做法来分割数据,当滚动条拖动接近底部（或右部）时，再去追加下一批所需要渲染的元素，每次滚动追加都会触发

    回流

    ,当追加到一定数量，触及浏览器瓶颈时，遗留的数据将会越来越多导致性能雪崩。
  • 我百万计数量的系统 给我十条一页的分，这不对吧
  1. 虚拟滑动
  • 这个方法只渲染可视范围内需要展示的内容，也就是图中的绿色区块，在可视区域之外的元素均可以不做渲染。
  • 原理是在滚动条滚动时动态更新列表项，所以只会渲染出计算出来的那部分内容，以及translate不会触发回流效果。

  • [注]：实际上考虑页面流畅性，不可能完全不渲染视区之外的内容，建议是预留2-3屏。

使用虚拟滑动
<template>
  <div class="scroll-dmeo">
    <!-- 滑动 可视区域 start -->
    <div @scroll="wrapperScroll($event)" ref="scrollRef" class="scroll-box">
      <div class="wrapper-scroll" :style="{ height: containerHeight + 'px' }" style="position: relative;" >                 
        <div :style="{ transform: `translateY(${scrollTopWrapper}px)` }"  style="position: absolute; width: 100%;"   >
           <div  v-for="(item, key) in showItem" :key="key"  class="scroll-item" :style="{height:itemHeight + 'px'}">
              {{item}}
           </div>        
        </div>   
      </div>
    </div>
    <!-- 滑动 可视区域 end -->
    
    <!-- 添加数据量按钮 start -->
    <div class="btns">
        <el-button @click="getList(100)" type="primary">+100</el-button>
        <el-button @click="getList(1000)" type="success">+1000</el-button>
        <el-button @click="getList(10000)" type="warning">+10000</el-button>
        <el-button @click="getList(100000)" type="danger">+100000</el-button>
        <p>{{allData.length}}</p>
    </div>
    <!-- 添加数据量按钮 end -->
  </div>
</template>

<script setup>
  import { reactive,ref,nextTick,computed } from 'vue';

  // ======================== 总数据量 ========================

  const allData = ref([]) //全部数据的集合 (相当于一次性获取到所有的列表数据)
  // 获取数据（num）表示数据量
  function getList(num){
    allData.value = [] //初始化数组
    for(let i = 0;i < num; i++){
      allData.value.push(i+1)
    }
  }

  // ======================== 可视容器计算 ========================

  // 获取可视区容器高度
  const scrollRef = ref(null)
  const scrollHeight = ref(0) //容器高度
  const showItemNum = ref(0) // 真是渲染DOM数量
  const itemHeight = 50  // item高度50px
  // 渲染完成后 计算容器高度 以及需要渲染的DOM数量
  nextTick(() => {  
    scrollHeight.value = scrollRef?.value?.clientHeight;
    console.log(`可视区域容器高度：${scrollHeight.value}px`);
    showItemNum.value = Math.ceil(scrollHeight.value / itemHeight);//真实DOM数 = Math.ceil(容器高度 / item高度)
    console.log(`视窗内应该显示的 DOM 数量：${showItemNum.value}个`);
  });
  //真实容器高度 = item总数 * 每条item的高度
  let containerHeight = ref(allData.value.length * itemHeight)

  // ======================== 渲染计算  ========================

  let startKey = ref(0); //当前状态的索引
  let scrollTopWrapper = ref(0); //片段容器偏移量

  // 滚动监听 使用防抖 防止滚动事件高频触发消耗内存资源 
  const  wrapperScroll = lodash.throttle((e) =>{
    //计算当前状态的索引  当前状态的索引 = Math.floor(当前滚动高度 / 每条item的高度);
    let tempNum = Math.floor(e.target.scrollTop / itemHeight); 
    //当前状态的索引发生变化才触发视图层刷新
    if (tempNum !== startKey.value) {    
      startKey.value = tempNum    
      scrollTopWrapper.value = e.target.scrollTop;  
    }
  },40)

  // 计算需要渲染item的数量  对数据进行切片处理方法
  const showItem = computed(() => { 
    return [...allData.value.slice(startKey.value, showItemNum.value + startKey.value + 3)];
  });
</script>

案例二

<template>
  <div>
    <el-button type="danger" @click="virtualScrolling(20)">20条</el-button>
    <el-button type="primary" @click="virtualScrolling(100)">一百条</el-button>
    <el-button type="success" @click="virtualScrolling(1000)">一千条</el-button>
    <el-button @click="virtualScrolling(100000)">十万条</el-button>
    <div class="wrap" @scroll="liScroll">
        <ul class="ul_wrap" :style="`height:${ulHei}px`">
            <li class="li_item" :style="`height:${liHei}px;transform:translateY(${ScroolNum}px)`"
                v-for="item in liList" :key="item">
                {{item}}
            </li>
        </ul>
    </div>
  </div>
</template>

<script>
export default{
    data(){
      return {
        liHei: 50,//li的高度
        ulHei: 480,//ul的高度
        liList: [],//真实展示的列表
        scrollHei:0,//@scroll事件滚动的top值
        ScroolNum: 0,//scrollHei能被li高度取余数的整数值。ScroolNum=scrollHei-(scrollHei%liHei)
        showList: 0,//真实展示的条数
        tableData: [],//全部数据的集合
        lastTime:0,//最后时间
      }
    },
    mounted () {
      this.virtualScrolling(100)
    },
    methods:{
      /**滚动监听 */
       liScroll (e) {
           if(new Date().getTime()-this.lastTime>40){//设置时间间隔，防止滚动事件高频触发消耗内存资源
           this.ele = e;//保存元素，方便重置scrollTop值
           this.scrollHei = e.target.scrollTop;//保存滚动条scrollTop值
           this.ScroolNum = this.scrollHei - (this.scrollHei % this.liHei);//获取已滚动到页面上方不可见的li元素的总高度（translateY的偏移高度）
           let len = this.ScroolNum / this.liHei;//计算已经有多少个li滚动到页面上方（视图上方用户不可见的数量）
           this.liList = this.tableData.slice(len, len + this.showList);//每次滚动事件后重新计算展示内容（截取的内容对应全部数据集的部分内容）
           this.lastTime=new Date().getTime();//记录最后一次更新时间
           }
           
       },
      /**初始化数据*/
       virtualScrolling (num) {
           let arr = [];//初始化数组
           for (let i = 0; i < num; i++) {//计算给定数据量
               arr.push(i+1)
           }
           this.tableData = arr;//全部数据集
           this.showList = Math.floor(this.ulHei / this.liHei) + 4;//计算真实渲染的列表数量
           this.liList = this.tableData.slice(0, this.showList);//初始化可视列表的内容
           this.lastTime=new Date().getTime();//记录最后一次更新时间
           if (!!this.ele) {//判断监听元素是否保存到ele字段中
               this.ele.target.scrollTop = 0;//如果元素存在ele中则将scrollTop初始化为0;
               this.ScroolNum=0;//初始化translateY的偏移高度
           }
       }
    }    
}
</script>
