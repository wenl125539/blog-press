了解 axios

  • axios 的参数配置(常用)
axios({
  method: 'get', // post、get、put、delete....
  baseURL: '', // 请求的域名，基本地址，公共的路径
  url: '', // 请求的路径
  params: {}, // get参数会将请求参数拼接在url上
  data: {}, // post会将请求参数放在请求体中
  headers: {}, // 设置请求头，例如设置token等
  timeout: 1000, // 设置请求超时时长，单位：ms
  withCredentials：default,//表明了是否是跨域请求、默认是default
  maxContentLength：1000//相应内容的最大值
})

  • baseURL 与 url 关系

    我们会请求某个域名下的多个接口，我们不希望每次发送请求都填写完整的 url，这时我们可以配置一个 baseURL，之后都可以传相对路径 axios使用正则判断是否已经存在域名 如果不是存在 则会继续拼接👇

// 例1 不存在域名
axios.create({
  baseURL: '/',
  url: '/api/login'
})
// 发出请求为当前运行的域名拼接 如果是本地运行 发出的请求则为https://localhost/api/login 因为经过代理所以同样能请求到服务器的接口数据

// 例2 不存在域名
axios.create({
  baseURL: '/AAA',
  url: '/api/login'
})
// 在上一步的基础上再拼接同样是运行环境域名 https://localhost/AAA/api/login

// 例3 存在域名
axios.create({
  baseURL: 'https://wenl.com',
  url: '/api/login'
})
// 发出请求为https://wenl.com/api/login

  • withCredentials 有什么用?

    跨域请求是否提供凭据信息(cookie、HTTP认证及客户端SSL证明等)，也可以简单的理解为，当前请求为跨域类型时是否在请求中协带cookie。 当配置了xhr.withCredentials = true时，必须在后端增加 response 头信息Access-Control-Allow-Origin，且必须指定域名，而不能指定为* 否则的话，请求会失败。

const instance = axios.create({
  baseURL: 'https://wenl.com',
})

instance.get('/api/login')

instance.post('/api/getInfo')
// 我们一旦配置了 baseURL，之后请求传入的 url 都会和我们的 baseURL 拼接成完整的绝对地址，除非请求传入的 url 已经是绝对地址。
// 所以此时真正发出的请求为 https://wenl.com/api/login

  • 请求拦截器

    在发送请求之前做些什么?

    一般用于给请求设置请求头 以及其他配置

// 创建实例
// axios.js
const instance = axios.create({
  timeout: 10000,
  baseURL: 'https://www.xxx.com', //接口域名
  withCredentials: true,
  crossDomain: true,
})
instance.interceptors.request.use((config) => {
    // 发出请求前做点啥？
    // 可以设置请求头 token 或者一些其他操作
    // 例如
    config.headers.me = '好人' //给请求头设置me是好人
    return config
  },
  (error) => {
    return Promise.reject(error)
  }
)

  • 响应拦截器

    接收到接口返回的结果 我们想要怎么处理?

    一般根据接口返回的状态码进行不同的处理

instance.interceptors.response.use(({ data }) => {
  // 状态码为200 则直接通过
  if (data.status === 200) {
    return data.data
  } else {
    // 其他状态 一律不通过
    return Promise.reject(data.msg)
  }
}, (error) => {
  if (axios.isCancel(error)) { // 取消请求的情况下，终端Promise调用链
    return new Promise(() => {})
  } else {
    return Promise.reject(error)
  }
})
// 这样就可以导出给其他地方使用了
export default instance

再封装一个处理get post参数的文件 @/utils/api.request.js

// @/utils/api.request.js
import instance from '@/utils/axios'
export default (config) => {
  const { url, method, data, headers, baseURL } = config
  const options = {
    url,
    method,
    headers
  }
  // 如果接收到baseURL 则使用传入的baseURL进行拼接 规则跟上面的baseURL例子一样
  if (baseURL) {
    options.baseURL = baseURL
  }
  // 根据请求方法 写入参数
  if (method) {
    switch (method.toLowerCase()) {
      case 'get' || 'delete':
        options.params = data
        break
      case 'post' || 'put':
        options.data = data
        break
    }
  }
  return instance.request(options)
}
