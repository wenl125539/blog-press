组件

<template>
  <div>
    <div @scroll="wrapperScroll($event)" :style="{height: scrollHeight+ 'px' }" style="overflow-y: scroll;position: relative;">
      <div :style="{ height: containerHeight + 'px' }" style="position: relative;">
        <div :style="{ transform: `translateY(${ScroolNum}px)` }" style="position: absolute; width: 100%;">
          <slot></slot>
        </div>
      </div>
    </div>
  </div>
</template>

<script>
export default {
  name: 'scrollList',
  props: {
    // 单个item的高度
    itemHeight: {
      type: Number,
      require: true
    },
    // 视窗高度
    scrollHeight: {
      type: Number,
      require: true
    },
    // 全部数据
    allData: {
      type: Array,
      require: true
    },
    // 列表改变是否改变滚动条位置
    isReset: {
      type: Boolean,
      default: true
    }
  },

  data() {
    return {
      ele: null,
      containerHeight: 0,
      scrollHei: 0, // 滚动条位置
      ScroolNum: 0, // 偏移量
      showList: 0, // 真实展示的条数
      liList: [], // 真实展示的列表
      lastTime: 0 // 最后时间 节流用
    }
  },

  watch: {
    allData() {
      this.virtualScrolling()
    }
  },

  methods: {
    // 监听滚动条位置
    wrapperScroll(e) {
      if (new Date().getTime() - this.lastTime > 40) { // 设置时间间隔，防止滚动事件高频触发消耗内存资源
        this.ele = e// 保存元素，方便重置scrollTop值
        this.scrollHei = e.target.scrollTop// 保存滚动条scrollTop值
        this.getList()
        this.lastTime = new Date().getTime()// 记录最后一次更新时间
      }
    },

    // 截取渲染列表
    getList() {
      this.ScroolNum = this.scrollHei - (this.scrollHei % this.itemHeight)// 获取已滚动到页面上方不可见的li元素的总高度（translateY的偏移高度）
      const len = this.ScroolNum / this.itemHeight// 计算已经有多少个li滚动到页面上方（视图上方用户不可见的数量）
      this.liList = this.allData.slice(len, len + this.showList)// 每次滚动事件后重新计算展示内容（截取的内容对应全部数据集的部分内容）
      this.$emit('input', this.liList)
    },

    // 初始化列表参数、视窗高度，滚动条位置
    virtualScrolling() {
      this.containerHeight = this.allData.length * this.itemHeight
      this.showList = Math.floor(this.scrollHeight / this.itemHeight) + 4// 计算真实渲染的列表数量 多加4条提前渲染（优化滑动体验）
      this.liList = this.allData.slice(0, this.showList)// 初始化可视列表的内容
      this.lastTime = new Date().getTime()// 记录最后一次更新时间
      this.isResetScroll()
      if (this.ScroolNum === 0) {
        this.$emit('input', this.liList)
      }
    },
    // 若数量减少 保持滚动条位置处于底部位置
    resetEndScroll() {
      this.scrollHei = this.containerHeight - this.scrollHeight
      this.ScroolNum = 0
      this.getList()
    },
    // 列表变化 滚动条位置
    isResetScroll() {
      if (!this.isReset) {
        // 如果列表数量减少了 ，最滚动条位置依旧居于最底部
        if (this.scrollHei > this.containerHeight) this.resetEndScroll()
      } else {
        if (this.ele) { // 判断监听元素是否保存到ele字段中
          this.ele.target.scrollTop = 0// 如果元素存在ele中则将scrollTop初始化为0;
          this.ScroolNum = 0// 初始化translateY的偏移高度
        }
      }
    }
  }
}
</script>

使用

  1. 给组件传四个必须参数
  • 双向绑定一个真实渲染的列表(循环的列表 案例中的liList)
  • 单个列的高度itemHeight
  • 可视滑动窗口的高度scrollHeight
  • 请求获取到的所有数据allData
  • 列表数据更新是否重置滚动条isReset,默认重置isReset=true

export default {
  // components: { ScrollList },
  data() {
    return {
      isReset:true,
      allData: [],
      itemHeight: 87, // item高
      scrollHeight: 500, // 视窗高度
      liList: [] // 真实展示的列表
    }
  },
  mounted() {
    this.add(500)
  },
  methods: {
    add(num) {
      this.allData = []
      for (let i = 0; i < num; i++) {
        this.allData.push({
          id: i,
          date: '2016-05-02',
          name: '王小虎',
          address: '上海市普陀区金沙江路 1518 弄'
        })
      }
    }
  }
}
