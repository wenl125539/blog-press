渲染第一个场景和物体

创建相机

【目标：知道基本使用流程】 透视相机

<template>
  <div ref="threeDay1Ref"  style="height:500px;" ></div>
</template>

<script setup>
  import { ref,nextTick } from 'vue'
  // import * as THREE from 'three'
// 1.创建场景
const scene = new THREE.Scene()

// 2.给场景添加物体
// 创建几何体  
// BoxGeometry(width : Float, height : Float, depth : Float, widthSegments : Integer, heightSegments : Integer, depthSegments : Integer)
// BoxGeometry(宽, 高, 深, 宽度分段数 , 高度分段数, 深度分段数)
const cubeGeometry = new THREE.BoxGeometry(1,1,2)
// 给物体添加材质
const cubMaterial = new THREE.MeshBasicMaterial({color:0x00ff00,wireframe:true})
// 根据几何体材质创建物体 THREE.Mesh(几何体，材质)
const cube = new THREE.Mesh(cubeGeometry,cubMaterial)
// 3.将几何体添加到场景中
scene.add(cube)

// 4.初始化渲染器
const renderer = new THREE.WebGLRenderer()
let threeDay1Ref = ref(null) //渲染位置的元素
// 确保加载完成后再获取(threeDay1Ref)元素内容
nextTick(() =>{
  // 5.将webgl渲染的canvsa内容添加到元素上
  threeDay1Ref.value.appendChild(renderer.domElement)
  // 设置渲染尺寸大小
  renderer.setSize(threeDay1Ref.value.clientWidth,threeDay1Ref.value.clientHeight)
  createCamera()
})

// 6.创建相机 （这里创建透视相机）
function createCamera() {
  // THREE.PerspectiveCamera(角度,屏幕宽高比,渲染距离（近端），渲染距离（远端）)
  const camera = new THREE.PerspectiveCamera(75,threeDay1Ref.value.clientWidth/threeDay1Ref.value.clientHeight,0.1,100)
  // 设置相机的位置 camera.position.set(X轴,Y轴,Z轴)
  camera.position.set(0,0,10)
  // 将相机添加到场景中（★）
  // scene.add(camera)
  // 使用渲染器,通过相机将场景渲染进来
  renderer.render(scene,camera)
}
</script>

requestAnimationFrame动画

time处理动画

requestAnimationFrame会默认给传入函数返回一个时间参数
注意：这个时间参数间隔并不是均匀的（受性能影响） 所以想要做均匀的动画效果 就要写其他计算方法 类似节流
eg:render(time)

function render(time){
  let t = time/1000
  console.log('时间间隔'+ t);
  renderer.render(scene,camera)
  requestAnimationFrame(render)
}

【目标：Clock跟踪时间使用】

可用于跟踪时间
