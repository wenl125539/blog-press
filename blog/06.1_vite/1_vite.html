二. 多环境配置

  • 多环境配置官网

.env

在根目录创建 .env | .env.XXX文件
在项目执行 npm run dev 的时候vite会自动去读取.env.development文件里面的配置
执行npm runbuild进行打包之后也会自动将.env.production的内容打包进去
这里注意的是，如果你想进入预发布模式的话需要在打包的时候进行mode配置:npm run build --mode staging

公共的 .env  
开发环境 .env.development  
生产环境 .env.production  
预发布环境 .env.staging

4. server相关【启动id端口、代理配置】

export default ({ command, mode }: ConfigEnv): UserConfig => {
  return {
    server:{
      //指定服务器应该监听哪个 IP 地址， 如果将此设置为 0.0.0.0 或者 true 将监听所有地址，包括局域网和公网地址
      // 默认：'127.0.0.1'  命令设置：--host 0.0.0.0 或 --host
      host: '127.0.0.1', 
      port: '3000', //指定开发服务器端口。注意：如果端口已经被使用，Vite 会自动尝试下一个可用的端口，所以这可能不是开发服务器最终监听的实际端口 默认  ：3000 | 5173
      strictPort: false , //'设为true时若端口已被占用则会直接退出，而不是尝试下一个可用端口 默认：false
      https:false, // 是否开启https
      open: false  // 项目启动就自动打开页面 【一般不作配置】 可以是true | false  [启动是否自动打开浏览器]   | '/public/index.html' 项目启动就自动打开这个index页面
      // 服务器代理
      proxy: {
        '/api': { // 匹配请求路径，localhost:3000/api
            target: 'http://wenl.com', // 代理的接口域名
            // secure: false,  // 如果是https接口，需要配置这个参数
            changeOrigin: false, // 开发模式，默认的origin是真实的 origin:localhost:3000 代理服务会把origin修改为目标地址
            // ws: true, // 是否代理websockets
            rewrite: (path) => path.replace(/^\/api/, '')
      }
    },
  }
}

5.构建相关【打包相关】

export default ({ command, mode }: ConfigEnv): UserConfig => {
  return {
      build:{
      // target:设置最终构建的浏览器兼容目标。
      // 默认：'modules'	指支持原生 ES 模块的浏览器。
      // 自定义目标也可以是一个 ES 版本（例如：es2015）、一个浏览器版本（例如：chrome58）或是多个目标组成的一个数组。
      target:'es2015',
      // cssTarget:只在针对非主流浏览器时使用。 最直观的示例是当你要兼容的场景是安卓微信中的 webview【可看官网介绍】
      outDir:'dist', // 指定输出路径（相对于 项目根目录). 默认：dist
      assetsDir:'assets', // 指定生成静态资源的存放路径（相对于 build.outDir） 默认：assets
      assetsInlineLimit:'4096', // 小于此阈值的导入或引用资源将内联为 base64 编码，以避免额外的 http 请求。
      cssCodeSplit: true, // css代码分割   true  在异步 chunk 中导入的 CSS 将内联到异步 chunk 本身，并在其被加载时插入。 | false 整个项目中的所有 CSS 将被提取到一个 CSS 文件中
      cssTarget: 'modules',// 默认值： 与 build.target 一致   自定义规则也是
      chunkSizeWarningLimit: 500 // chunk 大小警告的限制（以 kbs 为单位） 默认500
    }
  }
}
