组件(Vue2)

<template>
  <div>
    <div class="commodity" :style="{height:scrollHeight + 'px'}">
      <!--// 左侧导航内容 -->
      <div class="leftNav">
        <div ref="leftContant">
          <div class="leftEventList" v-for="(item, index) in scrollData" :key="index" :class="{ typeStyle: index === currentIndex }" @click="onClickNav(index)">
            <slot name="leftItem" :item="item"></slot>
          </div>
        </div>
      </div>
      <!--// 右侧商品内容 -->
      <div class="rightContant">
        <div ref="shopContant">
          <div class="eventList" v-for="(children, index) in scrollData" :key="index">
            <div class="groupName">{{children[groupName]}}</div>
            <div v-for="(item, index) in children[childName]" :key="index">
              <slot name="rightItem" :item="item"></slot>
            </div>
          </div>
        </div>
      </div>
    </div>
  </div>
</template>

<script>
import BScroll from "better-scroll"; //引入BScroll插件
export default {
  props: {
    // 滑动窗口高度
    scrollHeight: {
      type: Number,
      default: 500
    },
    leftHeight: {
      type: Number,
      default: 50
    },
    scrollData: {
      type: Array,
      default: []
    },
    // 分组的名称
    groupName: {
      type: String,
      require: true
    },
    // 分组对应列表的key
    childName: {
      type: String,
      require: true
    },
  },
  data() {
    return {
      leftScroll:'', //左边滑动
      leftSide: [], // 所有右侧分类li的top组成的数组  (列表第一次显示后就不再变化)
      foodsScroll:'', //右边滑动
      scrollY: 0, // 右侧滑动的时候距离顶部的值
      rightSide: [], // 所有右侧分类li的top组成的数组  (列表第一次显示后就不再变化)
      newIndex:0, // 当前选中的分组下标
      halfHeight:this.scrollHeight/2 // 视窗的一半
    };
  },
  mounted() {
    this.initializeRoll(); //滑动初始化方法
    this.initializeOwn(); //初始化rightSide方法
  },
  //计算属性
  computed: {
    // 计算出当前分类的下标
    currentIndex() {
      const { scrollY, rightSide} = this; 
      //拿到每次滑动的值以及每一个分类的下标值
      //findIndex() 方法返回传入一个测试条件（函数）符合条件的数组第一个元素位置
      //获取数组rightSide中每一个下标值并筛选条件返回第一个元素索引位置
      const index = rightSide.findIndex((top, index) => {
        // scrollY>=当前top && scrollY<下一个top
        return scrollY >= top && scrollY < rightSide[index + 1];
      });
      // 返回结果
      return index;
    }
  },
  watch:{
    // 分组的变化
    currentIndex(){
      // this.currentIndex>this.newIndex 表示正在向下滑  否则上滑
      if(this.currentIndex>this.newIndex){
        this.scrollLeft(this.currentIndex,-this.leftHeight)
      }else if(this.currentIndex<this.newIndex){
        this.scrollLeft(this.currentIndex,+ this.leftHeight)
      }
      this.newIndex = this.currentIndex
    }
  },
  methods: {

    // 初始化滚动 （监听滚动）
    initializeRoll() {
      // new一个类的实例，前提必须确保父元素和子元素的内容已经正确渲染了，不然不会生效
      this.leftScroll = new BScroll(".leftNav", {
        probeType: 3, //不仅在滚动实时的派发scroll事件，并且在滚动动画的时候也会派发scroll事件
        click: true
      });

      // 组件实例上挂属性
      this.foodsScroll = new BScroll(".rightContant", {
        probeType: 2, // 要滚动会实时的派发scroll事件
        click: true
      });

      // 给右侧列表绑定scroll监听
      this.foodsScroll.on("scroll", ({
        x,
        y
      }) => {
        this.scrollY = Math.abs(y);
      });

      // 给右侧列表绑定scroll结束的监听
      this.foodsScroll.on("scrollEnd", ({
        x,
        y
      }) => {
        // console.log("操作结束的下标值" + x, y);
        this.scrollY = Math.abs(y);
      });
    },

    // 确保item列渲染完成 再调用该方法进行初始化
    // 初始化坐标
    initializeOwn() {
      // 初始化leftSide
      const leftSide = [];
      let leftTop = 0;
      leftSide.push(leftTop);
      // 找到所有分类下的li
      const leftLis = this.$refs.leftContant.getElementsByClassName("leftEventList"); //获取dom元素 然后再获取到对应元素的位置 
      // slice() 方法返回一个新的数组对象
      Array.prototype.slice.call(leftLis).forEach(li => {
        leftTop += li.clientHeight; //相当于top = top + li.clientHeight
        leftSide.push(leftTop);
      });
      this.leftSide = leftSide


      // 初始化rightSide
      const rightSide = [];
      let top = 0;
      rightSide.push(top);
      // 找到所有分类下的li
      const lis = this.$refs.shopContant.getElementsByClassName("eventList"); //获取dom元素 然后再获取到对应元素的位置 
      // slice() 方法返回一个新的数组对象
      Array.prototype.slice.call(lis).forEach(li => {
        top += li.clientHeight; //相当于top = top + li.clientHeight
        rightSide.push(top);
      });
      //更新数据
      this.rightSide = rightSide;
    },
    
    //左侧点击事件
    onClickNav(index) {
      this.newIndex = index
      //拿到目标位置的scrollY，点击左侧导航，右侧列表滑动到对应的位置
      const scrollY = this.rightSide[index];
      // 点击的分类成为当前分类，索引当前选中的导航标题
      this.scrollY = scrollY;
      // 平滑滑动右侧列表
      this.foodsScroll.scrollTo(0, -scrollY, 1000);
    },

    // 左边 随 右边滚动
    scrollLeft(index,px){
      // 当前左边选中item的位置少于 滑动窗口的一半  直接滚动到顶部
      // 当前左边选中item的位置大于 滑动窗口的一半  直接滚动到底部
      // 中间位置的item, 则上下移动一个item的高度
      if(this.halfHeight > this.leftSide[index]){
        this.leftScroll.scrollTo(0, 0, 1000);
      }else if(this.leftSide[index] > this.leftSide[this.leftSide.length-1]-this.halfHeight){
          this.leftScroll.scrollTo(0, -(this.leftSide[this.leftSide.length-1] - this.scrollHeight), 1000);
      }else{
        this.leftScroll.scrollBy(0, px, 1000);
      }
    }
  }
};
</script>

<template>
  <div>
    <div class="scroll-box">
      <DoubleScroll :scrollHeight='500' :scrollData="scrollData" groupName='className' childName="students" :leftHeight="50">
        <!-- 左边栏 start -->
        <template #leftItem="data">
          <div class="class-name">{{data.item.className}}</div>
        </template>
        <!-- 左边栏 end -->
        
        <!-- 右边 start -->
        <template #rightItem="data">
          <div class="stu">
            <p>{{data.item.name}}</p>
            <p>{{data.item.age}}</p>
          </div>
        </template>
        <!-- 右边 end -->
      </DoubleScroll>
    </div>
  </div>
</template>

<script setup>
  import { reactive } from 'vue';
  const scrollData = [] //数据列表
  // 设置虚拟数据
  function getList() {
    for (let i = 0; i < 25; i++) {
      if(!scrollData[i]){scrollData[i] = {className:'',students:[]}}
      scrollData[i].className = `九年${i}班`
      for (let j = 0; j < 10; j++) {
        scrollData[i].students.push(
          {
            name:`学生${j}`,
            age:j+10
          }
        )
      }
    }  
  }
  getList()
  console.log(scrollData);

</script>
