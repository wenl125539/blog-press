@/store/modules/模块文件名XXX

// ========================index.js文件======================
// @/store/modules/模块文件名XXX/index.js
import state from './state'
import getters from './getters'
import mutations from './mutations'
import actions from './actions'

export default {
  namespaced: true, //重点 开启命名空间 避免重名
  state,
  getters,
  mutations,
  actions
}
// ========================state.js文件======================
// @/store/modules/模块文件名XXX/state.js
export default {
  msg: '' //示例
}
// ========================mutations.js文件==================
// @/store/modules/模块文件名XXX/mutations.js
export default {
  //示例
  setMsg(state, msg) {
    state.msg = msg
  }
  // ... 其他函数
}
// ========================actions.js文件====================
// @/store/modules/模块文件名XXX/actions.js
import { getMsgApi } from '@/api/modules/msg.js'
export default {
  // 获取接口msg 示例
  getMsg({state,commit}) {
    getMsgApi().then(res => {
      // 提交给当前模块的mutations的setMsg函数处理
      commit('setMsg', res)
      // 通知调用接口的地方 获取msg成功
      return Promise.resolve(res)
    }).catch((err) => {
      return Promise.reject(err)
    })
  }
  // ... 其他函数
}

使用辅助函数

mapState, mapGetters, mapMutations, mapActions

  • 单独构建版本的使用方式 （非多模块）
import { mapState , mapGetters, mapMutations , mapActions } from "vuex"
<script>
export default{
    computed:{
        ...mapState(["state变量A",'state变量B','...'']),
        ...mapGetters([
          'getters变量A',
          'getters变量B',
          // ...
        ])
    }    
    //还有一种使用方法,意思是将store里面的state里面的A属性映射到computed属性中的AA中。即AA代表了$store.state.A
    computed:{
        ...mapState({
            AA:"A"
        })
    },

    methods: {
      ...mapMutations([
        'mutations函数A', //  `this.mutations函数A()` 代表了 `this.$store.commit('mutations函数A')`
        'mutations函数B' //  `this.mutations函数B(参数)` 代表了 `this.$store.commit('mutations函数B', 参数)`
      ]),
      ...mapMutations({
        add: 'mutations函数C' //  `this.add()` 代表了 `this.$store.commit('mutations函数C')`
      })

      ...mapActions([
        'actions方法名A', // 将 `this.actions方法名A()` 映射为 `this.$store.dispatch('actions方法名A')`
        'actions方法名B' // 将 `this.actions方法名B(参数)` 映射为 `this.$store.dispatch('actions方法名B', 参数)`
      ]),
      ...mapActions({
        add: 'actions方法名C' // 将 `this.add()` 映射为 `this.$store.dispatch('actions方法名C')`
      })
  }
}
</script>

  • 多模块中使用辅助函数
import { mapState, mapGetters, mapMutations, mapActions } from 'vuex'
export default {
	computed: {
		...mapState('模块名', ['A', 'B', 'state变量'])，// mapState为例
		...mapGetters('模块名', ['A', 'B', 'getters变量']) // mapGetters为例
    // 具体到state变量里面某一属性
    ...mapState({
        B: state => state.模块名.变量A.变量A里面的B,
    })
	}，
	methods: {
		...mapMutations('模块名', ['A', 'B', 'mutations方法名'])，// mapMutations 为例
		...mapActions('模块名', ['A', 'B', 'actions方法名']) // mapActions 为例
	}
  // commit: 同步操作
  // 其他页面调用app.js 模块的mutations方法通过 this.$store.commit('模块名(非多模块可不填)/mutations方法名', 参数(可不填))
  // 例：其他页面调用app.js 模块的mutations方法通过 this.$store.commit('app/mutations方法名', 参数(可不填))
  // dispatch: 异步操作
  // 其他页面调用app.js 模块的action方法通过 this.$store.dispatch('模块名(非多模块可不填)/actions方法名', 参数(可不填))
  // 例：其他页面调用app.js 模块的mutations方法通过 this.$store.commit('app/actions方法名', 参数(可不填))
}

小技巧

State仓库字段太多怎么维护 —— 使用函数

// state有时候可能几十个字段，有些字段代表不一样的应用，我们可以使用函数来封装，返回一个对象，再使用 Object.assign合并对象
// 或
/*
* 合并对象的交集，以第一个对象为基础，将第二个对象的值合并到第一个对象中
* */
import _ from 'lodash'
export function assignExist(oldValue, newValue) {
  const tempOldValue = _.cloneDeep(oldValue)
  const tempNewValue = _.cloneDeep(newValue)
  _.forIn(tempOldValue, (value, key) => {
    if (_.has(tempNewValue, key)) {
      if (_.isObject(tempOldValue[key]) && !_.isArray(tempOldValue[key])) {
        tempOldValue[key] = assignExist(tempOldValue[key], tempNewValue[key])
      } else {
        tempOldValue[key] = tempNewValue[key]
      }
    }
  })
  return tempOldValue
}
