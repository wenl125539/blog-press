固定高度

<template>
  <RecycleScroller
    v-load="!loading"
    class="scroller"
    :items="list"
    :item-size="fontSize"
    key-field="id"
    :buffer="5000"
    :emitUpdate="true"
    @update="toUpdate"
  >
    <template v-slot="{ item }">
      <div class="goodbox" @click="toDetail(item.id)">
        {{item}}
      </div>
    </template>
    <template #after>
      <div class="mt-10 mb-20">
        <BottomLogo></BottomLogo>
      </div>
    </template>
  </RecycleScroller>
</template>
<script setup lang="ts">
import { ref, onMounted, reactive } from 'vue'
import BottomLogo from '@/components/BottomLogo.vue'
import type { PageType } from '@/types/page'
import { getGoodsList } from '@/api/modules/home'
import lodash from 'lodash'
import { useRouter } from 'vue-router'
import { GoodType } from '@/types/home'
const router = useRouter()
defineProps<{ isTop: boolean }>()
const loading = ref(false)

// RecycleScroller组件用到的itemSize是px单位，所以这里也需要动态计算出itemSize
// item需要渲染高度为52.4rem 再加上2rem顶部高度 所以每个item的高度为54.5rem
// px = rem * fontsize
const fontSize = ref(0) //一个item的高度
onMounted(() => {
  // 这里 就看是否需要转rem
  const htmlDoc = getComputedStyle(window.document.documentElement) as any
  fontSize.value = htmlDoc['font-size'].replace('px', '') * 54.5
  getList()
})

function update(start: any, end: any) {
  getList()
}
const toUpdate = lodash.debounce(update, 1000)

const isEnd = ref(false) //是否请求完数据
const page = reactive({
  pageSize: 10,
  currentPage: 1,
}) as PageType

const list = ref<GoodType[]>([])
// 请求列表数据
async function getList() {
  if (!isEnd.value) {
    try {
      // 调用接口
      const res = await getGoodsList(page)
      if (res) {
        // 判断如果是少于十条就不调接口
        list.value = list.value.concat(res.records)
        if (list.value.length >= res.totalSize) {
          isEnd.value = true
        } else {
          page.currentPage++
        }
      } else {
        isEnd.value = true
      }
      loading.value = true
    } catch (error) {
      console.log(error)
      loading.value = true
    }
  }
}

// 跳转到详情页
function toDetail(id: string) {
  router.push({
    path: '/goodDetail',
    query: {
      id,
    },
  })
}
</script>

非固定高度

DynamicScroller

<template>
  <div class="sellBox">
    <DynamicScroller
      :items="sellStore.saleList"
      :min-item-size="213"
      class="scroller"
      key-field="date"
      :style="{ overflow: isTop ? 'scroll' : 'hidden' }"
      :emitUpdate="true"
      :buffer="5000"
      @update="toUpdate"
      v-if="sellStore.saleList && sellStore.saleList.length > 0"
    >
      <template v-slot="{ item, index, active }">
        <DynamicScrollerItem
          watchData
          :item="item"
          :active="active"
          :data-index="index"
        >
          <!-- item样式 start-->
          <div class="leftLine"></div>
          {{item}}
          </div>
          <!-- item样式 end-->
        </DynamicScrollerItem>
      </template>
      <template #after>
        <div class="mt-10 mb-20">
          <BottomLogo></BottomLogo>
        </div>
      </template>
    </DynamicScroller>
  </div>
</template>
<script setup lang="ts">
import { onMounted } from 'vue'
import { saleCalendar } from '@/api/modules/home'
import BottomLogo from '@/components/BottomLogo.vue'
import { useSellStore } from '@/store/modules/sell'
import { useUserStore } from '@/store/modules/user'
import type { SellType } from '@/types/home'
import { showToast } from 'vant'
import lodash from 'lodash'
import { useRouter } from 'vue-router'
const router = useRouter()
const sellStore = useSellStore()
const useStore = useUserStore()
defineProps<{ isTop: boolean }>()

onMounted(() => {
  if (useStore.userInfo.saToken) {
    // 已订阅列表
    goodsSubscription()
  } else {
    // 未发售列表
    sellStore.getSaleCalendar()
  }
})
// 获取已订阅藏品的id
async function goodsSubscription() {
  await sellStore.goodsSubscribed()
  sellStore.getSaleCalendar()
}
// 触底
function update(start: any, end: any) {
  if (end < sellStore.saleList.length) {
    sellStore.lower()
  }
}
const toUpdate = lodash.debounce(update, 1000)
// 开售提醒
async function toRead(sale: SellType) {
  if (!useStore.userInfo.saToken) {
    router.push({ path: '/login', query: { toback: '1' } })
  } else {
    if (sale.isToBuy) {
      // 去购买
      // window.location.href = sale.buyUrl
      toDetail(sale.goodsId)
    } else if (sale.isRead) {
      showToast('你已订阅，无需再次订阅')
    } else {
      // 订阅
      const res = await saleCalendar({ goodsId: sale.goodsId })
      sellStore.checkIsRead(sale.goodsId)
      showToast('订阅成功')
    }
  }
}

// 跳转到详情页
function toDetail(id: string) {
  router.push({
    path: '/goodDetail',
    query: {
      id,
    },
  })
}
</script>
