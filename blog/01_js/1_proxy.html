proxy简介

只做简到不能再简的简介，详细用法请查阅MDN和阮一峰ES6

理解

Proxy 可以理解成，在目标对象之前架设一层“拦截”，外界对该对象的访问，都必须先通过这层拦截，因此提供了一种机制，可以对外界的访问进行过滤和改写。Proxy 这个词的原意是代理，用在这里表示由它来“代理”某些操作，可以译为“代理器”。

  • 语法
let p = new Proxy(target, handler);

水果代理商案代入理解

代理 理解简单案例
<template>
  <div>
    <xl-button type="info" @click="getfruit">获取{{isProxy?'代理后的':''}}水果</xl-button>
    <xl-button type="info" @click="getprice">获取价格price</xl-button>
    <xl-button type="primary" @click="setProxy">点击给水果找代理商（{{proxyNum}}）</xl-button>
    <xl-button type="warning" @click="resetFruit">重置水果</xl-button>
    <span>控制台查看</span>
    <br>
    <div v-show="isProxy">设置水果原价<input type="text" placeholder="给西瓜设置价格" :value="fruit.price" v-on:input="changePrice"></div>
  </div>
</template>

<script setup>
import {ref} from 'vue'
const isProxy = ref(false)
const proxyNum = ref(0)
let fruit = {
  name: "西瓜",
  price: 15,
};
//重置水果
function resetFruit() {
  fruit = {
    name: "西瓜",
    price: 15,
  };
  isProxy.value = false
  proxyNum.value = 0
  console.log("重置成功");
}
// 查看当前水果属性（看看是中间商的 还是果农的）
function getfruit() {
  console.log("fruit：", fruit);
}
function getName() {
  console.log("name：", fruit.name);
}
// 查看水果价格
function getprice() {
  console.log("price：", fruit.price);
}
// 给水果设置价格
function changePrice(event){
  fruit.price = Number.parseInt(event.target.value)
}
// ================核心代码start====================
// 给水果找代理商
function setProxy() {
  // 给水果创建一个代理 代理之后 只要是请求水果里面的属性都必须经过代理 
  fruit = new Proxy(fruit, {
    get(target, key) {
      let result = target[key];
      // 代理之后 需要加价格 （中间商嘛 肯定赚点差价呀）
      if (key === "price") result+=5;
      return result;
    },
    set(target, key, value) {
      // 给价格限制范围
      if (key === "price" &&  value >100) {
        throw Error("可不能太贵了");
      }
      console.log(`已设置价格为${value}`);
      return Reflect.set(target, key, value);
    },
  });
  isProxy.value = true
  proxyNum.value++
  console.log("代理成功");
}
// ================核心代码end====================
</script>
